# GitFolio Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use compose:subagent (recommended) or compose:execute to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build an Electron desktop app that scans a folder of git repos, lists all authors with commit counts, lets you select one or more, and exports a single markdown file of contributions with code diffs.

**Architecture:** electron-vite scaffolds the project; all git operations run in the Electron main process via `child_process.execFile`; preload exposes a typed API via `contextBridge`; renderer is a three-view React app (Scan → Authors → Export) with `useState` for navigation; progress events stream from main to renderer via `ipcRenderer.on`.

**Tech Stack:** Electron 36+, electron-vite 3+, React 19, TypeScript 5, Tailwind CSS 3, Vitest 2

import { contextBridge, ipcRenderer } from 'electron'
import type {
  ElectronAPI,
  ExportInput,
  RecentSelection,
  RepoInfo,
  ScanProgress,
  ExportProgress,
} from '../shared/types'

const api: ElectronAPI = {
  openFolderDialog: () => ipcRenderer.invoke('dialog:open-folder'),
  saveFileDialog: (defaultName) => ipcRenderer.invoke('dialog:save-file', defaultName),
  scanRepos: (folderPath) => ipcRenderer.invoke('repos:scan', folderPath),
  getAuthors: (repos: RepoInfo[]) => ipcRenderer.invoke('authors:get', repos),
  exportContributions: (input: ExportInput) => ipcRenderer.invoke('export:generate', input),
  getRecentSelections: () => ipcRenderer.invoke('prefs:get-recent'),
  saveRecentSelection: (selection: RecentSelection) =>
    ipcRenderer.invoke('prefs:save-recent', selection),

  onScanProgress(cb: (p: ScanProgress) => void) {
    const handler = (_: unknown, p: ScanProgress) => cb(p)
    ipcRenderer.on('scan:progress', handler)
    return () => ipcRenderer.removeListener('scan:progress', handler)
  },

  onExportProgress(cb: (p: ExportProgress) => void) {
    const handler = (_: unknown, p: ExportProgress) => cb(p)
    ipcRenderer.on('export:progress', handler)
    return () => ipcRenderer.removeListener('export:progress', handler)
  },
}

if (process.contextIsolated) {
  try {
    contextBridge.exposeInMainWorld('api', api)
  } catch (e) {
    console.error(e)
  }
} else {
  // @ts-ignore
  window.api = api
}

```typescript
import * as fs from 'fs'
import * as path from 'path'
import type { RepoInfo } from '../../shared/types'

export async function findGitRepos(
  folderPath: string,
  onProgress?: (currentPath: string) => void
): Promise<RepoInfo[]> {
  const repos: RepoInfo[] = []
  await walk(folderPath, repos, onProgress)
  return repos
}

async function walk(
  dir: string,
  repos: RepoInfo[],
  onProgress?: (currentPath: string) => void
): Promise<void> {
  let entries: fs.Dirent[]
  try {
    entries = await fs.promises.readdir(dir, { withFileTypes: true })
  } catch {
    return
  }

  if (entries.some((e) => e.isDirectory() && e.name === '.git')) {
    onProgress?.(dir)
    repos.push({ path: dir, name: path.basename(dir) })
    return
  }

  for (const entry of entries) {
    if (entry.isDirectory() && !entry.name.startsWith('.')) {
      await walk(path.join(dir, entry.name), repos, onProgress)
    }
  }
}

```typescript
import { execFile } from 'child_process'
import type { Author } from '../../shared/types'

export async function extractAuthorsFromRepo(repoPath: string): Promise<Author[]> {
  const stdout = await runGitLog(repoPath)
  if (!stdout.trim()) return []

  const counts = new Map<string, { name: string; count: number }>()
  for (const line of stdout.split('\n')) {
    if (!line.trim()) continue
    const sep = line.indexOf('\x1f')
    if (sep === -1) continue
    const name = line.slice(0, sep)
    const email = line.slice(sep + 1)
    const existing = counts.get(email)
    if (existing) {
      existing.count++
    } else {
      counts.set(email, { name, count: 1 })
    }
  }

  return Array.from(counts.entries()).map(([email, { name, count }]) => ({
    name,
    email,
    commitCount: count,
  }))
}

export function mergeAuthors(perRepoAuthors: Author[][]): Author[] {
  const totals = new Map<string, Author>()
  for (const repoAuthors of perRepoAuthors) {
    for (const author of repoAuthors) {
      const existing = totals.get(author.email)
      if (existing) {
        existing.commitCount += author.commitCount
      } else {
        totals.set(author.email, { ...author })
      }
    }
  }
  return Array.from(totals.values()).sort((a, b) => b.commitCount - a.commitCount)
}

function runGitLog(repoPath: string): Promise<string> {
  return new Promise((resolve) => {
    execFile(
      'git',
      ['log', '--all', '--format=%an\x1f%ae'],
      { cwd: repoPath, maxBuffer: 100 * 1024 * 1024, encoding: 'utf8' },
      (err, stdout) => {
        if (err) resolve('')
        else resolve(stdout)
      }
    )
  })
}

```typescript
import type { CommitDetail } from '../../shared/types'

export function generateMarkdown(authorName: string, commits: CommitDetail[]): string {
  if (commits.length === 0) {
    return `# Contribution Report: ${authorName}\n\nNo commits found.\n`
  }

  const dates = commits.map((c) => c.date).sort()
  const dateRange = `${dates[0]} → ${dates[dates.length - 1]}`
  const repoNames = new Set(commits.map((c) => c.repoName))

  const lines: string[] = [
    `# Contribution Report: ${authorName}`,
    '',
    `- **Total Repos:** ${repoNames.size}`,
    `- **Total Commits:** ${commits.length.toLocaleString()}`,
    `- **Date Range:** ${dateRange}`,
    '',
    '---',
    '',
  ]

  const byRepo = new Map<string, CommitDetail[]>()
  for (const c of commits) {
    const list = byRepo.get(c.repoName) ?? []
    list.push(c)
    byRepo.set(c.repoName, list)
  }

  for (const [repoName, repoCommits] of byRepo) {
    lines.push(`## Repo: ${repoName}`, '')
    const sorted = [...repoCommits].sort((a, b) => b.date.localeCompare(a.date))
    for (const c of sorted) {
      lines.push(...formatCommit(c))
    }
    lines.push('---', '')
  }

  return lines.join('\n')
}

function formatCommit(c: CommitDetail): string[] {
  const match = c.subject.match(/^(\w+)(?:\([^)]*\))?:\s*(.*)$/)
  const subjectLine = match ? `**${match[1]}:** ${match[2]}` : c.subject

  const lines: string[] = [
    `### ${c.date} — ${c.hash.slice(0, 7)}`,
    subjectLine,
    '',
  ]

  for (const file of c.files) {
    lines.push(`#### \`${file.path}\``)
    if (file.skipped) {
      lines.push(`_skipped: ${file.skipReason}_`, '')
      continue
    }
    lines.push('```diff')
    lines.push(file.diff)
    lines.push('```')
    if (file.truncated) {
      lines.push('<!-- ... truncated ... -->')
    }
    lines.push('')
  }

  return lines
}

```typescript
import { execFile } from 'child_process'
import type { CommitDetail, FileDiff } from '../../shared/types'

const LOCKFILES = new Set([
  'package-lock.json',
  'yarn.lock',
  'pnpm-lock.yaml',
  'Podfile.lock',
  'Gemfile.lock',
  'poetry.lock',
  'composer.lock',
  'Cargo.lock',
])

const MAX_DIFF_LINES = 200

export async function getCommitsWithDiffs(
  repoPath: string,
  repoName: string,
  authorEmails: string[],
  onProgress?: (hash: string) => void
): Promise<CommitDetail[]> {
  const authorArgs = authorEmails.flatMap((e) => ['--author', e])
  const logOut = await runGit(
    ['log', '--all', ...authorArgs, '--format=%H\x1f%an\x1f%ae\x1f%ad\x1f%s', '--date=short'],
    repoPath
  )
  if (!logOut.trim()) return []

  const commits: CommitDetail[] = []
  for (const line of logOut.split('\n')) {
    if (!line.trim()) continue
    const parts = line.split('\x1f')
    if (parts.length < 5) continue
    const [hash, authorName, authorEmail, date, ...rest] = parts
    const subject = rest.join('\x1f')
    if (!hash || !date) continue

    onProgress?.(hash)

    const rawDiff = await runGit(
      ['show', hash, '--patch', '--no-color', '--format='],
      repoPath
    )
    const files = parseDiff(rawDiff)
    const totalLinesChanged = files.reduce((sum, f) => sum + f.diff.split('\n').length, 0)

    commits.push({ hash, authorName, authorEmail, date, subject, repoName, repoPath, files, totalLinesChanged })
  }

  return commits
}

function parseDiff(rawDiff: string): FileDiff[] {
  const segments = rawDiff.split(/^diff --git /m).filter(Boolean)
  return segments.map((seg) => {
    const firstLine = seg.split('\n')[0]
    const match = firstLine.match(/ b\/(.+)$/)
    const filePath = match?.[1] ?? firstLine.trim()
    const diffContent = seg.split('\n').slice(1).join('\n')

    const fileName = filePath.split('/').pop() ?? filePath
    if (LOCKFILES.has(fileName)) {
      return { path: filePath, diff: '', truncated: false, skipped: true, skipReason: 'lockfile' }
    }

    if (diffContent.includes('Binary files')) {
      return { path: filePath, diff: '', truncated: false, skipped: true, skipReason: 'binary' }
    }

    const lines = diffContent.split('\n')
    if (lines.length > MAX_DIFF_LINES) {
      return {
        path: filePath,
        diff: lines.slice(0, MAX_DIFF_LINES).join('\n'),
        truncated: true,
        skipped: false,
      }
    }

    return { path: filePath, diff: diffContent, truncated: false, skipped: false }
  })
}

function runGit(args: string[], cwd: string): Promise<string> {
  return new Promise((resolve) => {
    execFile(
      'git',
      args,
      { cwd, maxBuffer: 500 * 1024 * 1024, encoding: 'utf8' },
      (err, stdout) => {
        if (err) resolve('')
        else resolve(stdout)
      }
    )
  })
}

import * as fs from 'fs'
import * as path from 'path'
import type { RecentSelection } from '../../shared/types'

const MAX_RECENT = 5

interface PrefsData {
  recentSelections: RecentSelection[]
}

export class PrefsStore {
  private readonly filePath: string
  private data: PrefsData

  constructor(userDataPath: string) {
    this.filePath = path.join(userDataPath, 'prefs.json')
    this.data = this.load()
  }

  getRecentSelections(): RecentSelection[] {
    return this.data.recentSelections
  }

  saveRecentSelection(selection: RecentSelection): void {
    const without = this.data.recentSelections.filter(
      (s) => s.folderPath !== selection.folderPath
    )
    this.data.recentSelections = [selection, ...without].slice(0, MAX_RECENT)
    this.persist()
  }

  private load(): PrefsData {
    try {
      return JSON.parse(fs.readFileSync(this.filePath, 'utf8'))
    } catch {
      return { recentSelections: [] }
    }
  }

  private persist(): void {
    try {
      fs.writeFileSync(this.filePath, JSON.stringify(this.data, null, 2))
    } catch (e) {
      console.error('Failed to save prefs:', e)
    }
  }
}

  })

    return findGitRepos(folderPath, (currentPath) => {

        phase: 'scanning',
        current: 0,
        total: 0,
        currentPath,
      })
    })
  })

      repos.map((repo, i) => {

          phase: 'reading-authors',
          current: i + 1,
          total: repos.length,
          currentPath: repo.name,

      })
    )
    return mergeAuthors(results)
  })

  ipcMain.handle('prefs:save-recent', (_event, selection) =>

### Task 11: ScanView

**Covers:** Acceptance Criteria 1, 2, 6 (scanning + author reading + progress)

**Files:**
- Create: `src/renderer/src/views/ScanView.tsx`

- [ ] **Step 1: Create src/renderer/src/views/ScanView.tsx**

```tsx
import { useState } from 'react'
import type { Author, RepoInfo, ScanProgress } from '../../../shared/types'
import { FolderDrop } from '../components/FolderDrop'
import { ProgressBar } from '../components/ProgressBar'

interface Props {
  onComplete: (folderPath: string, repos: RepoInfo[], authors: Author[]) => void
}

type Phase = 'idle' | 'scanning' | 'reading-authors'

export function ScanView({ onComplete }: Props) {
  const [phase, setPhase] = useState<Phase>('idle')
  const [progress, setProgress] = useState<ScanProgress | null>(null)
  const [repoCount, setRepoCount] = useState(0)
  const [error, setError] = useState<string | null>(null)

  async function handleFolder(folderPath: string) {
    setError(null)
    setPhase('scanning')
    setProgress(null)

    const unsub = window.api.onScanProgress((p) => {
      setProgress(p)
      if (p.phase === 'reading-authors') setPhase('reading-authors')
    })

    try {
      const repos = await window.api.scanRepos(folderPath)
      setRepoCount(repos.length)

      if (repos.length === 0) {
        unsub()
        setPhase('idle')
        setError('No git repositories found in that folder.')
        return
      }

      const authors = await window.api.getAuthors(repos)
      unsub()
      onComplete(folderPath, repos, authors)
    } catch (e: any) {
      unsub()
      setError(e.message ?? 'Scan failed.')
      setPhase('idle')
    }
  }

  if (phase === 'idle') {
    return (
      <div className="mt-12">
        <h2 className="text-2xl font-semibold mb-2">Pick a folder</h2>
        <p className="text-gray-400 mb-8">
          Select a directory containing your git repositories.
        </p>
        <FolderDrop onFolder={handleFolder} />
        {error && <p className="mt-4 text-red-400 text-sm">{error}</p>}
      </div>
    )
  }

  return (
    <div className="mt-12">
      <h2 className="text-2xl font-semibold mb-6">
        {phase === 'scanning' ? 'Scanning repositories…' : 'Reading authors…'}
      </h2>
      <ProgressBar
        current={progress?.current ?? 0}
        total={progress?.total || repoCount || 1}
        label={
          progress?.currentPath
            ? `${phase === 'scanning' ? 'Found' : 'Reading'}: ${progress.currentPath}`
            : undefined
        }
      />
      {phase === 'reading-authors' && repoCount > 0 && (
        <p className="mt-4 text-gray-400 text-sm">
          Found {repoCount} {repoCount === 1 ? 'repo' : 'repos'}, reading authors in parallel…
        </p>
      )}
    </div>
  )
}

```tsx
import { useState, useMemo, useEffect } from 'react'
import type { Author } from '../../../shared/types'
import { AuthorList } from '../components/AuthorList'

interface Props {
  authors: Author[]
  folderPath: string
  onExport: (emails: string[], savePath: string) => void
  onBack: () => void
}

export function AuthorsView({ authors, folderPath, onExport, onBack }: Props) {
  const [query, setQuery] = useState('')
  const [selected, setSelected] = useState<Set<string>>(new Set())
  const [saving, setSaving] = useState(false)

  useEffect(() => {
    window.api.getRecentSelections().then((recents) => {
      const match = recents.find((r) => r.folderPath === folderPath)
      if (match) setSelected(new Set(match.authorEmails))
    })
  }, [folderPath])

  const filtered = useMemo(() => {
    const q = query.trim().toLowerCase()
    if (!q) return authors
    return authors.filter(
      (a) => a.name.toLowerCase().includes(q) || a.email.toLowerCase().includes(q)
    )
  }, [authors, query])

  function toggle(email: string) {
    setSelected((prev) => {
      const next = new Set(prev)
      next.has(email) ? next.delete(email) : next.add(email)
      return next
    })
  }

  async function handleExport() {
    if (selected.size === 0) return
    setSaving(true)
    const emails = Array.from(selected)
    const defaultName = `gitfolio-${emails[0].split('@')[0]}.md`
    const savePath = await window.api.saveFileDialog(defaultName)
    setSaving(false)
    if (!savePath) return

    await window.api.saveRecentSelection({
      folderPath,
      authorEmails: emails,
      timestamp: Date.now(),
    })

    onExport(emails, savePath)
  }

  return (
    <div>
      <div className="flex items-center gap-4 mb-6">
        <button
          onClick={onBack}
          className="text-gray-400 hover:text-gray-200 text-sm transition-colors"
        >
          ← Back
        </button>
        <h2 className="text-2xl font-semibold">Select Authors</h2>
        <span className="text-gray-500 text-sm">{authors.length} found</span>
      </div>

      <input
        type="text"
        placeholder="Search by name or email…"
        value={query}
        onChange={(e) => setQuery(e.target.value)}
        className="w-full bg-gray-800 border border-gray-700 rounded-lg px-4 py-2.5 text-sm mb-4 focus:outline-none focus:border-indigo-500 transition-colors"
      />

      <div className="bg-gray-900 rounded-xl border border-gray-800 max-h-96 overflow-y-auto mb-6">
        {filtered.length === 0 ? (
          <p className="px-4 py-10 text-center text-gray-500 text-sm">
            No authors match your search.
          </p>
        ) : (
          <AuthorList authors={filtered} selected={selected} onToggle={toggle} />
        )}
      </div>

      <div className="flex items-center justify-between">
        <span className="text-sm text-gray-400">
          {selected.size === 0
            ? 'Select at least one author'
            : `${selected.size} ${selected.size === 1 ? 'author' : 'authors'} selected`}
        </span>
        <button
          disabled={selected.size === 0 || saving}
          onClick={handleExport}
          className="bg-indigo-600 hover:bg-indigo-500 disabled:opacity-40 disabled:cursor-not-allowed text-white font-medium px-6 py-2.5 rounded-lg transition-colors"
        >
          {saving ? 'Choose save location…' : 'Export Contributions →'}
        </button>
      </div>
    </div>
  )
}

```tsx
import { useEffect, useState } from 'react'
import type { ExportProgress, RepoInfo } from '../../../shared/types'
import { ProgressBar } from '../components/ProgressBar'

interface Props {
  repos: RepoInfo[]
  authorEmails: string[]
  savePath: string
  onDone: () => void
}

type State = 'running' | 'done' | 'error'

export function ExportView({ repos, authorEmails, savePath, onDone }: Props) {
  const [exportState, setExportState] = useState<State>('running')
  const [progress, setProgress] = useState<ExportProgress>({ current: 0, total: repos.length })
  const [error, setError] = useState<string | null>(null)

  useEffect(() => {
    const unsub = window.api.onExportProgress((p) => setProgress(p))

    window.api
      .exportContributions({ repos, authorEmails, savePath })
      .then(() => { unsub(); setExportState('done') })
      .catch((e: any) => {
        unsub()
        setError(e?.message ?? 'Export failed.')
        setExportState('error')
      })

    return unsub
  }, [])

  if (exportState === 'running') {
    return (
      <div className="mt-12">
        <h2 className="text-2xl font-semibold mb-6">Exporting contributions…</h2>
        <ProgressBar
          current={progress.current}
          total={progress.total}
          label={progress.currentRepo ? `Processing: ${progress.currentRepo}` : undefined}
        />
      </div>
    )
  }

  if (exportState === 'error') {
    return (
      <div className="mt-12">
        <h2 className="text-2xl font-semibold mb-4 text-red-400">Export failed</h2>
        <p className="text-gray-400 mb-6">{error}</p>
        <button
          onClick={onDone}
          className="bg-gray-700 hover:bg-gray-600 text-white px-6 py-2.5 rounded-lg transition-colors"
        >
          Start over
        </button>
      </div>
    )
  }

  return (
    <div className="mt-12">
      <div className="text-5xl mb-4">✅</div>
      <h2 className="text-2xl font-semibold mb-2">Done!</h2>
      <p className="text-gray-400 mb-2">Your contribution report has been saved to:</p>
      <p className="bg-gray-900 border border-gray-800 rounded-lg px-4 py-3 mb-8 text-indigo-400 font-mono text-sm break-all">
        {savePath}
      </p>
      <button
        onClick={onDone}
        className="bg-indigo-600 hover:bg-indigo-500 text-white font-medium px-6 py-2.5 rounded-lg transition-colors"
      >
        Export another →
      </button>
    </div>
  )
}
